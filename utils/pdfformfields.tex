\NewDocumentCommand{\FormField}{O{\linewidth} O{0.8cm} O{} m m}{
    % #1: width (optional)
    % #2: height (optional)
    % #3: options for TextField (e.g., multiline=true) -> Now unused
    % #4: caption/label
    % #5: field name for hyperref -> Now unused
    \begin{tcolorbox}[
        enhanced jigsaw,
        width=#1,
        height=#2,
        boxsep=0pt,
        left=2pt, right=2pt, top=2pt, bottom=2pt,
        colback=white,
        colframe=sheetaccent!75!black,
        boxrule=0.6pt,
        opacityback=0.68,
        opacityframe=0.85,
        valign=top,
        overlay={
            \node[anchor=north west, text=sheetaccent!85!black, font=\scriptsize\titlefont, text width=\dimexpr#1-8pt\relax, inner sep=0pt] at ([xshift=4pt, yshift=-4pt]frame.north west) {#4};
        }
    ]
    \vphantom{Abc} % to ensure the box has some content and correct height if empty
    \end{tcolorbox}
}
