\NewDocumentCommand{\FormField}{O{\linewidth} O{0.8cm} O{} m m}{
    % #1: width (optional)
    % #2: height (optional)
    % #3: options for TextField (e.g., multiline=true) -> Now unused
    % #4: caption/label
    % #5: field name for hyperref -> Now unused
    \begin{tcolorbox}[
        enhanced jigsaw,
        width=#1,
        height=#2,
        boxsep=0pt,
        left=2pt, right=2pt, top=2pt, bottom=2pt,
        colback=white,
        colframe=black!50,
        opacityback=0.2,
        opacityframe=0.5,
        valign=top,
        overlay={
            \node[anchor=north west, text=black!40, font=\small\titlefont, text width=\dimexpr#1-8pt\relax, inner sep=0pt] at ([xshift=4pt, yshift=-5pt]frame.north west) {#4};
        }
    ]
    \vphantom{Abc} % to ensure the box has some content and correct height if empty
    \end{tcolorbox}
}
