function get_i18n_string(key)
    if not ACTIVE_DATASET then
        tex.sprint("ERROR:NO_ACTIVE_DATASET")
        return
    end
    
    if ACTIVE_DATASET[key] and ACTIVE_DATASET[key][LANG] then
        tex.sprint(ACTIVE_DATASET[key][LANG])
    elseif i18n_data["all"][key] then
        tex.sprint(i18n_data["all"][key][LANG])
    else
        tex.sprint("UNDEFINED_I18N_KEY:'" .. key .. "'")
    end
end
\end{luacode*}
\newcommand{\trans}[1]{\directlua{get_i18n_string("#1")}}

\begin{luacode*}
function get_lang()
    tex.sprint(LANG)
end
\end{luacode*}
\newcommand{\lang}{\directlua{get_lang()}}

\begin{luacode*}
function get_babel_lang_name()
    local babel_map = { de = "german", en = "english" }
    tex.sprint(babel_map[LANG] or "english")
    print("==> I18N: Babel language name " .. babel_map[LANG] or "english")
end
